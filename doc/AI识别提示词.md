def cluster_lines(raw_lines):
    if not raw_lines: return []
    raw_lines.sort(key=lambda w: -w['box'][3])
    
    clusters = []
    while raw_lines:
        seed = raw_lines.pop(0)
        c_box = list(seed['box'])
        
        changed = True
        while changed:
            changed = False
            rest = []
            for item in raw_lines:
                # è®¡ç®—é—´éš™
                dx = max(0, c_box[0] - item['box'][2], item['box'][0] - c_box[2])
                dy = max(0, c_box[1] - item['box'][3], item['box'][1] - c_box[3])
                
                if dx < CONFIG['THRESHOLD_X_GAP'] and dy < CONFIG['THRESHOLD_Y_GAP']:
                    c_box[0] = min(c_box[0], item['box'][0])
                    c_box[1] = min(c_box[1], item['box'][1])
                    c_box[2] = max(c_box[2], item['box'][2])
                    c_box[3] = max(c_box[3], item['box'][3])
                    changed = True
                else:
                    rest.append(item)
            raw_lines = rest
        
        w, h = c_box[2]-c_box[0], c_box[3]-c_box[1]
        if w > 100 and h > 100:
            clusters.append({'box': c_box, 'w': w, 'h': h, 'src': 'GEO', 'w_set': [], 'h_set': []})
        
    return clusters

def extract_dim_value(dim_entity):
    text = dim_entity.dxf.text
    if text and text.strip():
        if "%%d" in text or "Â°" in text: return 0, "ANGLE"
        if "<>" in text: return cad_round(dim_entity.dxf.actual_measurement), "MEA(<>)"
        txt = re.sub(r'\\[A-Za-z0-9]+;', '', text) 
        txt = re.sub(r'\{.*?\}', '', txt)         
        txt = txt.replace(';', '').replace(r'\P', '')
        match = re.search(r'(\d+)', txt)
        if match: return int(match.group(1)), "TXT"
    return cad_round(dim_entity.dxf.actual_measurement), "MEA"

def is_point_in_box(x, y, box):
    return (x >= box[0] and x <= box[2] and
            y >= box[1] and y <= box[3])

def is_box_inside_box(inner, outer):
    tol = 10.0
    return (inner[0] >= outer[0] - tol and
            inner[1] >= outer[1] - tol and
            inner[2] <= outer[2] + tol and
            inner[3] <= outer[3] + tol)

def remove_table_frames(candidates, anchor_x, anchor_y):
    valid_windows = []
    for w in candidates:
        if is_point_in_box(anchor_x, anchor_y, w['box']):
            continue
        valid_windows.append(w)
    return valid_windows

def clean_overlaps(windows):
    if not windows: return []
    windows.sort(key=lambda w: -(w['w'] * w['h'])) 
    final = []
    for w in windows:
        is_sub = False
        for big in final:
            if is_box_inside_box(w['box'], big['box']):
                is_sub = True
                break
        if not is_sub:
            final.append(w)
    return final

def sort_windows_into_grid(windows):
    if not windows: return []
    windows.sort(key=lambda w: -w['box'][3])
    rows = []
    current_row = [windows[0]]
    current_y = windows[0]['box'][3]
    for w in windows[1:]:
        y = w['box'][3]
        if abs(y - current_y) < 500:
            current_row.append(w)
        else:
            current_row.sort(key=lambda item: item['box'][0])
            rows.append(current_row)
            current_row = [w]
            current_y = y
    current_row.sort(key=lambda item: item['box'][0])
    rows.append(current_row)
    return rows

def select_best_value(geo_val, candidates):
    pool = [d for d in candidates if d.val > CONFIG['MIN_VAL_FILTER']]

    for d in pool:
        if abs(d.val - geo_val) < 150: return d.val, "EXACT"
    vals = [d.val for d in pool]
    return max(vals), "MAX"

                    curr[3] = max(curr[3], other[3])
